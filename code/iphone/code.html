Code for playing music


import UIKit
import AVFoundation

class ViewController: UIViewController {

    @IBOutlet weak var imageView: UIImageView!
        
    var ding:AVAudioPlayer = AVAudioPlayer()
    
    override func viewDidLoad() {
        super.viewDidLoad()

        // add images to the array
        var imagesListArray : [UIImage] = []
        // use for loop
        for position in 1...16
        {
            
            let strImageName : String = "win_\(position).png"
            let image  = UIImage(named:strImageName)
            imagesListArray.append(image!)
        }
        
        imageView.animationImages = imagesListArray
        imageView.animationDuration = 1.0
        imageView.startAnimating()
        
        // play song
        prepareAudios()
        ding.play()
    }

    func prepareAudios() {
        let filePath = NSBundle.mainBundle().pathForResource("music", ofType: "mp3")
        if let path = filePath {
            let sound = NSURL.fileURLWithPath(path)
            do {
                ding = try AVAudioPlayer(contentsOfURL: sound)
                ding.prepareToPlay()
            }
            catch {
                print("Error: Can't play song")
            }
        }
        else {
            print("Error: Can't find file")
        }
    }
}
