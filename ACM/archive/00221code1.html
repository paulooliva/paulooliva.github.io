#include <stdio.h>
#include <stdlib.h>

typedef struct{
  double left, right, top, bottom, height;
  int n;
} Building;

Building b[105];
int bsize;

int cmp_b(Building *a, Building *b){
  if(a->left < b->left) return -1;
  else if(a->left > b->left) return 1;
  else if(a->bottom < b->bottom) return -1;
  else return 1;
}

int isVisible(int i){
  int x;
  int first = 1;
  double cleft, cright;
  for(x = 0; x < bsize && b[x].left <= b[i].right; x++){
    if(x == i) continue;
    if(b[x].right < b[i].left) continue;
    if(b[x].height < b[i].height) continue;
    if(b[x].bottom > b[i].bottom) continue;
    if(first){
      cleft = b[x].left;
      cright = b[x].right;
      first = 0;
    } else {
      if(b[x].left > cright) break;
      cright = cright > b[x].right ? cright : b[x].right;
      if(cright >= b[i].right) break;
    }
  }

  if(first) return 1;
  if(cleft <= b[i].left && cright >= b[i].right) return 0;
  return 1;
}

int main(){
  int i;
  int first;
  int mapnum = 1;

  while(scanf("%d", &bsize) == 1 && bsize){
    for(i = 0; i < bsize; i++){
      scanf("%lf %lf %lf %lf %lf", &b[i].left, &b[i].bottom, &b[i].right, &b[i].top, &b[i].height);
      b[i].n = i+1;
      b[i].right += b[i].left;
      b[i].top += b[i].bottom;
    }

    qsort(b, bsize, sizeof(b[0]), (void *) cmp_b);
    if(mapnum > 1) printf("\n");
    printf("For map #%d, the visible buildings are numbered as follows:\n", mapnum++);

    first = 1;
    for(i = 0; i < bsize; i++){
      if(isVisible(i)){
        /*if(first) first = 0;
          else printf(" ");*/
        printf("%d ", b[i].n);
      }
    }
    printf("\n");
  }
  return 0;
}
