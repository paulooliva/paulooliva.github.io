#include 

typedef struct{
  int N, E, S, W;
} Box;


Box grid[10][10];
int n;

void ReadLine(){
  char d;
  int i, j;

  scanf("%c %d %d ", &d, &i, &j);
  if(d == 'H'){
    grid[i][j].N = 1;
    grid[i-1][j].S = 1;
  } else {
    grid[j][i].W = 1;
    grid[j-1][i].E = 1;
  }

}

void ClearGrid(){
  int i, j;

  for(i = 0; i < 10; i++)
    for(j = 0; j < 10; j++){
     grid[i][j].N = 0;
     grid[i][j].E = 0;
     grid[i][j].W = 0;
     grid[i][j].S = 0;
    }
}
int CheckRow(int row, int col, int size){
  int i;

  for(i = col; i < col+size; i++){
    if(grid[row][i].N != 1) return 0;
  }
  return 1;
}

int CheckCol(int row, int col, int size){
  int i;
  for(i = row; i < row+size; i++){
    if(grid[i][col].W != 1) return 0;
  }
  return 1;
}

int CheckSize(int size){
  int row, col;
  int count = 0;

  for(row = 1; row <= n; row++)
    for(col = 1; col <= n; col++){
      if( CheckRow(row, col, size) &&
          CheckRow(row+size, col, size) &&
          CheckCol(row, col, size) &&
          CheckCol(row, col+size, size)) count++;
    }
  return count;
}

void Process(){
  int size;
  int found = 0;
  int count;

  for(size = 1; size <= n; size++){
    count = CheckSize(size);
    if(count != 0){
      found = 1;
      printf("%d square (s) of size %d\n", count, size);
    }
  }
  if(found == 0){
    printf("No completed squares can be found.\n");
  }

  printf("\n");

}

int main(){
  int pnum = 0;
  int m;
  int i;

  while(scanf(" %d ", &n) == 1){
   if(pnum != 0){
     printf("**********************************\n");
     printf("\n");
   }
    scanf(" %d ", &m);
    pnum++;

    ClearGrid();
    for(i = 0; i < m; i++)
      ReadLine();

    printf("Problem #%d\n", pnum);
    printf("\n");

    Process();
  }
  return 0;
}
