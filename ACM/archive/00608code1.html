#include <stdio.h>

int main(){
  char left[3][10], right[3][10], res[3][10];
  int loc[3][12], tnum, i, j, coin, light, found;

  scanf("%d", &tnum);
  while(tnum--){
    for(i = 0; i < 3; i++){
      for(j = 0; j <  12; j++) loc[i][j] = 3;
      scanf("%s %s %s", left[i], right[i], res[i]);
      for(j = 0; j < strlen(left[i]); j++)  loc[i][(int)(left[i][j]-'A')] = 1;
      for(j = 0; j < strlen(right[i]); j++) loc[i][(int)(right[i][j]-'A')] = 2;
    }
    for(coin = 0; coin < 12; coin++){
      for(light = 0; light <= 1; light++){
        found = 1;
        for(i = 0; i < 3; i++){
          switch(loc[i][coin]){
          case 3:
            if(res[i][0] != 'e') found = 0;
            break;
          case 1:
            if((light && res[i][0] != 'd') ||
               (!light && res[i][0] != 'u')) found = 0;
            break;
          case 2:
            if((light && res[i][0] != 'u') ||
               (!light && res[i][0] != 'd')) found = 0;
            break;
          }
          if(!found) break;
        }
        if(found){
          printf("%c is the counterfeit coin and it is %s.\n",
                 'A'+coin, light ? "light" : "heavy");
          goto end;
        }
      }
    }
  end:
  }
  return 0;
}
