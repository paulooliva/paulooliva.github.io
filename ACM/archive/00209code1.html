/* Solution: Find the x,y,z coordinates and then just count the number of occurances along the x, y, z -
could simplify this by writing this generally instead of just for each figure. */

#include <stdio.h>
#include <math.h>

typedef struct{
  int n, x, y, z;
} Point;

Point p[10];
int psize;

int isTriangle(){
  if(p[0].x == p[1].x){
    if((p[0].y == p[2].y && p[1].z == p[2].z) ||
       (p[0].z == p[2].z && p[1].y == p[2].y)) return 1;
  } else if(p[0].y == p[1].y){
    if((p[0].x == p[2].x && p[1].z == p[2].z) ||
       (p[0].z == p[2].z && p[1].x == p[2].x)) return 1;

  } else if(p[0].z == p[1].z){
    if((p[0].y == p[2].y && p[1].x == p[2].x) ||
       (p[0].x == p[2].x && p[1].y == p[2].y)) return 1;
  }
  return 0;
}

int isQuad(){
  int xs[4], xsize=0;
  int ys[4], ysize=0;
  int zs[4], zsize=0;
  int i, j, m;

  for(i = 0; i < 4; i++){
    m = 0;
    for(j = 0; j < xsize; j++)
      if(xs[j] == p[i].x){
        m = 1;
        break;
      }
    if(!m) xs[xsize++] = p[i].x;
    m = 0;
    for(j = 0; j < ysize; j++)
      if(ys[j] == p[i].y){
        m = 1;
        break;
      }
    if(!m) ys[ysize++] = p[i].y;
    m = 0;
    for(j = 0; j < zsize; j++)
      if(zs[j] == p[i].z){
        m = 1;
        break;
      }
    if(!m) zs[zsize++] = p[i].z;
  }
  if(xsize == 3 && ysize == 2 && zsize == 2) return 1;
  if(xsize == 2 && ysize == 3 && zsize == 2) return 1;
  if(xsize == 2 && ysize == 2 && zsize == 3) return 1;
  return 0;
}

int isHex(){
  int xs[6], xsize=0;
  int ys[6], ysize=0;
  int zs[6], zsize=0;
  int i, j, m;

  for(i = 0; i < 6; i++){
    m = 0;
    for(j = 0; j < xsize; j++)
      if(xs[j] == p[i].x){
        m = 1;
        break;
      }
    if(!m) xs[xsize++] = p[i].x;
    m = 0;
    for(j = 0; j < ysize; j++)
      if(ys[j] == p[i].y){
        m = 1;
        break;
      }
    if(!m) ys[ysize++] = p[i].y;
    m = 0;
    for(j = 0; j < zsize; j++)
      if(zs[j] == p[i].z){
        m = 1;
        break;
      }
    if(!m) zs[zsize++] = p[i].z;
  }
  if(xsize == 3 && ysize == 3 && zsize == 3) return 1;
  return 0;
}

int main(){
  int pset[10],  tsize, i, j, dupe, x;

  char buff[100];

  while(fgets(buff, 100, stdin)){
    psize = 0;

    tsize = sscanf(buff, "%d %d %d %d %d %d", &pset[0], &pset[1], &pset[2], &pset[3], &pset[4], &pset[5]);
    for(i = 0; i < tsize; i++){
      if(i != 0) printf(" ");
      printf("%d", pset[i]);
      dupe = 0;
      for(j = 0; j < psize; j++){
        if(p[j].n == pset[i]){
          dupe = 1;
          break;
        }
      }
      if(!dupe){
        p[psize].n = pset[i];
        x = sqrt(2*pset[i])-1;
        while(1){
          if(x*(x-1)/2 < pset[i] && pset[i] <= x*(x+1)/2) break;
          x++;
        }
        p[psize].x = x;
        p[psize].y = pset[i]-x*(x-1)/2;
        p[psize].z = x*(x+1)/2+1 - pset[i];
        psize++;
      }
    }
    switch(psize){
    case 3:
      if(isTriangle()) printf(" are the vertices of a triangle\n");
      else printf(" are not the vertices of an acceptable figure\n");
      break;
    case 4:
      if(isQuad()) printf(" are the vertices of a parallelogram\n");
      else printf(" are not the vertices of an acceptable figure\n");
      break;
    case 6:
      if(isHex()) printf(" are the vertices of a hexagon\n");
      else printf(" are not the vertices of an acceptable figure\n");
      break;
    default:
      printf(" are not the vertices of an acceptable figure\n");
    }
  }
  return 0;
}
